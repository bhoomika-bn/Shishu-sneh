## 1. PROJECT OVERVIEW

### 1.1 Purpose
Shishu-Sneh is a healthcare companion app for new mothers to track their baby's first-year growth, vaccinations, feeding, and developmental milestones—especially targeting rural Indian mothers with limited healthcare access.

### 1.2 Core Features
- **Baby Profile Management**: Single baby profile with DOB, birth weight, gender
- **Growth Tracking**: Monthly weight/height with WHO-standard percentile charts
- **Vaccination Scheduler**: Auto-populated India NIS schedule with smart reminders
- **Feeding Guide**: Daily AI-generated breastfeeding & nutrition tips
- **Milestone Tracker**: Weekly developmental checklists (Yes/Not Yet/Skip)
- **Multilingual Support**: 3+ Indian languages
- **Offline-First**: All core features work without internet
- **Data Export**: PDF summary for sharing

### 1.3 Tech Constraints
- **Platform**: Android only (min SDK 24, target SDK 34)
- **Language**: Kotlin preferred
- **Architecture**: MVVM + Repository Pattern
- **Database**: Room DB (SQLite)
- **AI**: Gemini API for tips & chatbot

### 🔐 **SCREEN 1: SPLASH SCREEN**

#### Purpose
- Check if user is logged in
- Check if baby profile exists
- Navigate to appropriate screen

#### UI Elements
```kotlin
@Composable
fun SplashScreen(navController: NavController, viewModel: SplashViewModel) {
    val navigationState by viewModel.navigationState.collectAsState()
    
    LaunchedEffect(Unit) {
        viewModel.checkAuthStatus()
    }
    
    LaunchedEffect(navigationState) {
        when (navigationState) {
            is NavigationState.ToHome -> navController.navigate("home") {
                popUpTo("splash") { inclusive = true }
            }
            is NavigationState.ToLogin -> navController.navigate("login") {
                popUpTo("splash") { inclusive = true }
            }
            is NavigationState.ToOnboarding -> navController.navigate("onboarding") {
                popUpTo("splash") { inclusive = true }
            }
            else -> { /* Loading */ }
        }
    }
    
    Box(
        modifier = Modifier.fillMaxSize(),
        contentAlignment = Alignment.Center
    ) {
        Column(horizontalAlignment = Alignment.CenterHorizontally) {
            Image(
                painter = painterResource(R.drawable.logo),
                contentDescription = "Shishu-Sneh Logo",
                modifier = Modifier.size(120.dp)
            )
            Spacer(Modifier.height(16.dp))
            Text("Shishu-Sneh", style = MaterialTheme.typography.headlineMedium)
            Spacer(Modifier.height(8.dp))
            CircularProgressIndicator()
        }
    }
}
```

#### ViewModel Logic
```kotlin
@HiltViewModel
class SplashViewModel @Inject constructor(
    private val babyRepository: BabyRepository,
    private val authRepository: AuthRepository
) : ViewModel() {
    
    private val _navigationState = MutableStateFlow<NavigationState>(NavigationState.Loading)
    val navigationState: StateFlow<NavigationState> = _navigationState
    
    fun checkAuthStatus() {
        viewModelScope.launch {
            delay(2000) // Splash duration
            
            val isLoggedIn = authRepository.isUserLoggedIn()
            if (!isLoggedIn) {
                _navigationState.value = NavigationState.ToLogin
                return@launch
            }
            
            val hasBabyProfile = babyRepository.hasBabyProfile()
            _navigationState.value = if (hasBabyProfile) {
                NavigationState.ToHome
            } else {
                NavigationState.ToOnboarding
            }
        }
    }
}

sealed class NavigationState {
    object Loading : NavigationState()
    object ToLogin : NavigationState()
    object ToOnboarding : NavigationState()
    object ToHome : NavigationState()
}
```

```kotlin
@Composable
fun LoginScreen(navController: NavController, viewModel: LoginViewModel) {
    var email by remember { mutableStateOf("") }
    var password by remember { mutableStateOf("") }
    val loginState by viewModel.loginState.collectAsState()
    
    Column(
        modifier = Modifier
            .fillMaxSize()
            .padding(24.dp),
        horizontalAlignment = Alignment.CenterHorizontally,
        verticalArrangement = Arrangement.Center
    ) {
        Image(
            painter = painterResource(R.drawable.logo),
            contentDescription = null,
            modifier = Modifier.size(100.dp)
        )
        
        Spacer(Modifier.height(32.dp))
        
        OutlinedTextField(
            value = email,
            onValueChange = { email = it },
            label = { Text("Email or Phone") },
            modifier = Modifier.fillMaxWidth(),
            keyboardOptions = KeyboardOptions(keyboardType = KeyboardType.Email)
        )
        
        Spacer(Modifier.height(16.dp))
        
        OutlinedTextField(
            value = password,
            onValueChange = { password = it },
            label = { Text("Password") },
            modifier = Modifier.fillMaxWidth(),
            visualTransformation = PasswordVisualTransformation(),
            keyboardOptions = KeyboardOptions(keyboardType = KeyboardType.Password)
        )
        
        Spacer(Modifier.height(24.dp))
        
        Button(
            onClick = { viewModel.login(email, password) },
            modifier = Modifier.fillMaxWidth(),
            enabled = loginState !is LoginState.Loading
        ) {
            if (loginState is LoginState.Loading) {
                CircularProgressIndicator(color = Color.White, modifier = Modifier.size(24.dp))
            } else {
                Text("Login")
            }
        }
        
        Spacer(Modifier.height(16.dp))
        
        TextButton(onClick = { navController.navigate("register") }) {
            Text("Don't have an account? Register")
        }
        
        // Handle login states
        when (val state = loginState) {
            is LoginState.Success -> {
                LaunchedEffect(Unit) {
                    if (state.hasBabyProfile) {
                        navController.navigate("home") {
                            popUpTo("login") { inclusive = true }
                        }
                    } else {
                        navController.navigate("onboarding") {
                            popUpTo("login") { inclusive = true }
                        }
                    }
                }
            }
            is LoginState.Error -> {
                Text(state.message, color = MaterialTheme.colorScheme.error)
            }
            else -> {}
        }
    }
}
```

```kotlin
// Chart update logic
fun updateChart(chart: LineChart, records: List<GrowthRecord>, isWeight: Boolean) {
    val entries = records.mapIndexed { index, record ->
        Entry(
            index.toFloat(),
            if (isWeight) record.weightKg else record.heightCm
        )
    }
    
    val dataSet = LineDataSet(entries, if (isWeight) "Weight" else "Height").apply {
        color = Color.Blue.toArgb()
        valueTextSize = 12f
        lineWidth = 2f
        setCircleColor(Color.Blue.toArgb())
        circleRadius = 4f
    }
    
    // Add WHO percentile lines (simplified - real implementation needs WHO data)
    val percentile50 = LineDataSet(
        records.mapIndexed { index, _ ->
            Entry(index.toFloat(), calculateWHO50thPercentile(index))
        },
        "WHO 50th"
    ).apply {
        color = Color.Green.toArgb()
        lineWidth = 1f
        enableDashedLine(10f, 5f, 0f)
        setDrawCircles(false)
    }
    
    val lineData = LineData(dataSet, percentile50)
    chart.data = lineData
    chart.invalidate()
}
```

```kotlin
    private val growthRecordDao: GrowthRecordDao,

    fun getAllRecords(babyId: Long): Flow<List<GrowthRecord>> {
        return growthRecordDao.getAllRecordsForBaby(babyId)

    suspend fun addGrowthRecord(weightKg: Float, heightCm: Float) {
        val baby = babyRepository.babyProfile.first() ?: return
        
        val ageInWeeks = calculateAgeInWeeks(baby.dateOfBirth)
        val percentileWeight = WHOGrowthCalculator.calculatePercentile(
            ageInWeeks, weightKg, baby.gender, isWeight = true
        )
        val percentileHeight = WHOGrowthCalculator.calculatePercentile(
            ageInWeeks, heightCm, baby.gender, isWeight = false
        )
        
        val record = GrowthRecord(
            babyId = baby.id,
            recordDate = System.currentTimeMillis(),
            weightKg = weightKg,
            heightCm = heightCm,
            ageInWeeks = ageInWeeks,
            whoPercentileWeight = percentileWeight,
            whoPercentileHeight = percentileHeight
        )
        
        growthRecordDao.insert(record)
    }
}
```

### 4.3 WHO Growth Calculator (Simplified)

```kotlin
object WHOGrowthCalculator {
    
    // Simplified - Real implementation needs WHO LMS tables
    fun calculatePercentile(
        ageInWeeks: Int,
        value: Float,
        gender: String,
        isWeight: Boolean
    ): Float {
        // This is a simplified placeholder
        // Real implementation should use WHO LMS data:
        // https://www.who.int/tools/child-growth-standards/standards
        
        val standardData = getWHOStandard(ageInWeeks, gender, isWeight)
        val zScore = (value - standardData.median) / standardData.sd
        
        return zScoreToPercentile(zScore)
    }
    
    private data class WHOStandard(val median: Float, val sd: Float)
    
    private fun getWHOStandard(weeks: Int, gender: String, isWeight: Boolean): WHOStandard {
        // Placeholder - load from assets or hardcoded table
        return if (isWeight) {
            when (weeks) {
                0 -> WHOStandard(3.3f, 0.4f)
                4 -> WHOStandard(4.5f, 0.5f)
                8 -> WHOStandard(5.7f, 0.6f)
                12 -> WHOStandard(6.4f, 0.7f)
                else -> WHOStandard(7.0f, 0.8f)
            }
        } else {
            when (weeks) {
                0 -> WHOStandard(49.0f, 2.0f)
                4 -> WHOStandard(54.0f, 2.2f)
                8 -> WHOStandard(59.0f, 2.4f)
                12 -> WHOStandard(62.0f, 2.5f)
                else -> WHOStandard(65.0f, 2.6f)
            }
        }
    }
    
    private fun zScoreToPercentile(z: Float): Float {
        // Standard normal CDF approximation
        return (0.5 * (1 + erf(z / sqrt(2f)))) * 100f
    }
}
```

```kotlin
@Singleton
class GeminiRepository @Inject constructor(
    @ApplicationContext private val context: Context
) {
    private val generativeModel = GenerativeModel(
        modelName = "gemini-1.5-flash",
        apiKey = BuildConfig.GEMINI_API_KEY
    )
    
    suspend fun generateDailyFeedingTip(babyAgeWeeks: Int): String {
        val prompt = """
            You are a pediatric nutrition expert. Generate a short, practical feeding tip 
            for a ${babyAgeWeeks}-week-old baby in India. Focus on:
            - Exclusive breastfeeding for first 6 months
            - Mother's nutrition
            - Breastfeeding techniques
            - Common myths to avoid
            
            Keep the tip under 100 words, warm, and culturally appropriate for Indian mothers.
            Do NOT give medical advice - only general wellness tips.
        """.trimIndent()
        
        return try {
            val response = generativeModel.generateContent(prompt)
            response.text ?: "Exclusive breastfeeding is best for your baby's first 6 months."
        } catch (e: Exception) {
            Log.e("GeminiRepository", "Error generating tip", e)
            getFallbackTip(babyAgeWeeks)
        }
    }
    
    suspend fun answerFeedingQuestion(question: String, babyAgeWeeks: Int): String {
        val prompt = """
            You are a supportive pediatric advisor. A mother asks about her ${babyAgeWeeks}-week-old baby:
            
            Question: $question
            
            Provide a helpful, concise answer (under 150 words) that:
            - Is evidence-based and aligned with WHO/Indian Academy of Pediatrics guidelines
            - Is warm and encouraging
            - Suggests consulting a doctor if the question is medical in nature
            - Avoids jargon and is culturally sensitive for Indian context
        """.trimIndent()
        
        return try {
            val response = generativeModel.generateContent(prompt)
            response.text ?: "I recommend consulting your pediatrician for personalized advice."
        } catch (e: Exception) {
            Log.e("GeminiRepository", "Error answering question", e)
            "I'm sorry, I couldn't process your question. Please consult your doctor."
        }
    }
    
    suspend fun getMilestoneEncouragement(
        milestone: String,
        achieved: Boolean,
        babyAgeWeeks: Int
    ): String {
        val prompt = if (achieved) {
            "Generate a short congratulatory message (30 words) for a parent whose ${babyAgeWeeks}-week-old baby just achieved: $milestone"
        } else {
            "Generate a short encouraging message (30 words) for a parent whose ${babyAgeWeeks}-week-old baby hasn't yet achieved: $milestone. Emphasize that all babies develop at their own pace."
        }
        
        return try {
            val response = generativeModel.generateContent(prompt)
            response.text ?: if (achieved) "Great milestone!" else "Every baby is unique!"
        } catch (e: Exception) {
            if (achieved) "Wonderful progress!" else "Keep encouraging your baby!"
        }
    }
    
    private fun getFallbackTip(weeks: Int): String {
        val tips = mapOf(
            0..8 to "Exclusive breastfeeding is the best nutrition for your newborn. Feed on demand, day and night.",
            9..16 to "Breast milk provides all the nutrients your baby needs. Ensure you're well-hydrated and eating nutritious food.",
            17..24 to "Continue breastfeeding. Your baby may feed more frequently during growth spurts - this is normal!",
            25..52 to "Exclusive breastfeeding until 6 months is recommended by WHO. Your breast milk adapts to your baby's needs."
        )
        
        return tips.entries.first { weeks in it.key }.value
    }
}
```

### 5.2 ViewModel Integration

```kotlin
@HiltViewModel
class FeedingViewModel @Inject constructor(
    private val geminiRepository: GeminiRepository,
    private val babyRepository: BabyRepository,
    private val feedingTipDao: FeedingTipDao
) : ViewModel() {
    
    private val _todayTip = MutableStateFlow<FeedingTip?>(null)
    val todayTip: StateFlow<FeedingTip?> = _todayTip
    
    private val _chatHistory = MutableStateFlow<List<ChatMessage>>(emptyList())
    val chatHistory: StateFlow<List<ChatMessage>> = _chatHistory
    
    private val _isLoadingResponse = MutableStateFlow(false)
    val isLoadingResponse: StateFlow<Boolean> = _isLoadingResponse
    
    init {
        loadTodayTip()
    }
    
    private fun loadTodayTip() {
        viewModelScope.launch {
            val baby = babyRepository.babyProfile.first() ?: return@launch
            val ageWeeks = calculateAgeInWeeks(baby.dateOfBirth)
            
            // Check if we have today's tip
            val today = LocalDate.now()
            val existingTip = feedingTipDao.getTipForDate(today.toEpochDay())
            
            if (existingTip != null) {
                _todayTip.value = existingTip
            } else {
                // Generate new tip
                val tipContent = geminiRepository.generateDailyFeedingTip(ageWeeks)
                val newTip = FeedingTip(
                    babyId = baby.id,
                    date = today.toEpochDay(),
                    content = tipContent,
                    babyAgeWeeks = ageWeeks
                )
                feedingTipDao.insert(newTip)
                _todayTip.value = newTip
            }
        }
    }
    
    fun askQuestion(question: String) {
        viewModelScope.launch {
            val baby = babyRepository.babyProfile.first() ?: return@launch
            val ageWeeks = calculateAgeInWeeks(baby.dateOfBirth)
            
            // Add user message
            _chatHistory.value = _chatHistory.value + ChatMessage(question, isUser = true)
            _isLoadingResponse.value = true
            
            // Get AI response
            val answer = geminiRepository.answerFeedingQuestion(question, ageWeeks)
            
            // Add AI response
            _chatHistory.value = _chatHistory.value + ChatMessage(answer, isUser = false)
            _isLoadingResponse.value = false
        }
    }
}

data class ChatMessage(
    val content: String,
    val isUser: Boolean,
    val timestamp: Long = System.currentTimeMillis()
)
```
